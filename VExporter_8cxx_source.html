122 void XmlVGM::VExporter::ProcessPositions(VGM::IVolume* volume)
123 {
125 
126  std::string volumeName = volume->Name();
127 
128  // store the name of logical volume in the set
129  fVolumeNames.insert(fVolumeNames.begin(), volumeName);
130 
131  int nofDaughters = volume->NofDaughters();
132 
133  if (nofDaughters > 0) {
134  for (int i = 0; i < nofDaughters; i++) {
135 
136  // Simple placement positions
137  //
138  VGM::IPlacement* dPlacement = volume->Daughter(i);
139  if (dPlacement->Type() == VGM::kSimplePlacement) {
140 
141  VGM::Transform transform = dPlacement->Transformation();
142  std::string name = fMaps.AddPosition(transform);
143 
144  if (name != "") fWriter->WritePosition(name, transform);
145  }
146 
147  // Positions in tessellated solids
148  //
149  VGM::ISolid* dSolid = dPlacement->Volume()->Solid();
150  if (dSolid->Type() == VGM::kTessellated) {
151 
152  ProcessPositionsInTessellated(dSolid);
153  }
154 
155  // Positions in multi union solids
156  //
157  if (dSolid->Type() == VGM::kMultiUnion) {
158 
159  ProcessPositionsInMultiUnion(dSolid);
160  }
161  /*
162  // Displacemnt positions in Boolean solids
163  //
164  VGM::ISolid* dSolid = dPlacement->Volume()->Solid();
165  if (dSolid->Type() == VGM::kBoolean) {
166 
167  ProcessPositionsInBoolean(dSolid);
168  }
169  */
170 
171  std::string dVolumeName = dPlacement->Volume()->Name();
172  if (fVolumeNames.find(dVolumeName) == fVolumeNames.end()) {
173  // process volumed only if it was not yet processed
174  ProcessPositions(dPlacement->Volume());
175  }
176  }
177  }
178 }

181 void XmlVGM::VExporter::ProcessPositionsInBoolean(VGM::ISolid* solid)
182 {
185 
186  if (solid->Type() != VGM::kBoolean) return;
187 
188  VGM::IBooleanSolid* boolSolid = dynamic_cast<VGM::IBooleanSolid*>(solid);
189 
190  VGM::Transform transform = boolSolid->Displacement();
191  std::string name = fMaps.AddPosition(transform);
192 
193  // if (name != "")
194  // fWriter->WritePosition(name, transform);
195 
196  // Process constituents
197  VGM::ISolid* solidA = boolSolid->ConstituentSolidA();
198  VGM::ISolid* solidB = boolSolid->ConstituentSolidB();
199 
200  if (solidA->Type() == VGM::kBoolean) ProcessPositionsInBoolean(solidA);
201  if (solidB->Type() == VGM::kBoolean) ProcessPositionsInBoolean(solidB);
202 }

205 void XmlVGM::VExporter::ProcessPositionsInTessellated(VGM::ISolid* solid)
206 {
209 
210  if (solid->Type() != VGM::kTessellated) return;
211 
212  VGM::ITessellatedSolid* tessellatedSolid =
213  dynamic_cast<VGM::ITessellatedSolid*>(solid);
214 
215  for (int i = 0; i < tessellatedSolid->NofFacets(); ++i) {
216  for (int j = 0; j < tessellatedSolid->NofVertices(i); ++j) {
217 
218  VGM::ThreeVector vertex = tessellatedSolid->Vertex(i, j);
219  std::string name = fMaps.AddPosition(vertex);
220 
221  if (name != "") fWriter->WritePosition(name, vertex);
222  }
223  }

336 void XmlVGM::VExporter::ProcessMaterials(VGM::IVolume* volume)
337 {
339 
340  std::string volumeName = volume->Name();
341 
342  // Material
343  const VGM::IMaterial* volumeMaterial =
344  fFactory->MaterialFactory()->Material(volume->MaterialName());
345 
346  if (!volumeMaterial) {
347  std::cerr << "XmlVGM::VExporter::ProcessMaterials: " << std::endl;
348  std::cerr << " Material " << volume->MaterialName() << "not found."
349  << std::endl;
350  exit(1);
351  }
352 
353  const VGM::IMaterial* material = fMaps.AddMaterial(volumeMaterial);
354 
355  // Elements
356  if (material) {
357  for (int j = 0; j < int(material->NofElements()); j++) {
358  VGM::IElement* element = material->Element(j);
359  fMaps.AddElement(element);
360 
361  // Isotopes
362  for (int k = 0; k < int(element->NofIsotopes()); k++)
363  fMaps.AddIsotope(element->Isotope(k));
364  }
365  }
366 
367  // store the name of logical volume in the set
368  fVolumeNames.insert(fVolumeNames.begin(), volumeName);
369 
370  int nofDaughters = volume->NofDaughters();
371 
372  if (nofDaughters > 0) {
373  for (int i = 0; i < nofDaughters; i++) {
374 
375  VGM::IPlacement* dPlacement = volume->Daughter(i);
376 
377  std::string dVolumeName = dPlacement->Volume()->Name();
378  if (fVolumeNames.find(dVolumeName) == fVolumeNames.end()) {
379  // process volume only if it was not yet processed
380  ProcessMaterials(dPlacement->Volume());
381  }
382  }
383  }
384 }

435  // store the name of logical volume in the set

437 
438  // process daughters

441  for (int i = 0; i < nofDaughters; i++) {
442

445  << std::endl;

447 
448  VGM::IVolume* dVolume = volume->Daughter(i)->Volume();
449  std::string dVolumeName = dVolume->Name();
450 
451  if (fVolumeNames.find(dVolumeName) == fVolumeNames.end()) {
