197  if (Debug() > 0) {

201  std::cout << solid->GetName() << std::endl;

203 
204  G4VSolid* consSolid = solid;
205 
206  // Get constituent solid if the solid is a reflected solid
207  //
208  G4ReflectedSolid* reflSolid = dynamic_cast<G4ReflectedSolid*>(solid);
209  if (reflSolid) consSolid = reflSolid->GetConstituentMovedSolid();
210 
211  G4Box* box = dynamic_cast<G4Box*>(consSolid);
212  if (box) {
213  return Register(new Geant4GM::Box(box, reflSolid));
214  }
215 
216  G4Cons* cons = dynamic_cast<G4Cons*>(consSolid);
217  if (cons) {
218  return Register(new Geant4GM::Cons(cons, reflSolid));
219  }
220 
221  G4CutTubs* ctubs = dynamic_cast<G4CutTubs*>(consSolid);
222  if (ctubs) {
223  return Register(new Geant4GM::Ctubs(ctubs, reflSolid));
224  }
225 
226  G4Ellipsoid* ellipsoid = dynamic_cast<G4Ellipsoid*>(consSolid);
227  if (ellipsoid) {
228  return Register(new Geant4GM::Ellipsoid(ellipsoid, reflSolid));
229  }
230 
231  G4EllipticalTube* eltu = dynamic_cast<G4EllipticalTube*>(consSolid);
232  if (eltu) {
233  return Register(new Geant4GM::EllipticalTube(eltu, reflSolid));
234  }
235 
236  G4ExtrudedSolid* xtru = dynamic_cast<G4ExtrudedSolid*>(consSolid);

242  if (hype) {

247  if (para) {
248  return Register(new Geant4GM::Para(para, reflSolid));
249  }
250 
251  G4Paraboloid* paraboloid = dynamic_cast<G4Paraboloid*>(consSolid);
252  if (paraboloid) {
253  return Register(new Geant4GM::Paraboloid(paraboloid, reflSolid));

258  return Register(new Geant4GM::Polycone(polycone, reflSolid));
259  }
260

263  return Register(new Geant4GM::Polyhedra(polyhedra, reflSolid));

267  if (sphere) {
268  return Register(new Geant4GM::Sphere(sphere, reflSolid));

270

272  if (tessel) {

274  }
275 
276  G4Torus* torus = dynamic_cast<G4Torus*>(consSolid);
277  if (torus) {
278  return Register(new Geant4GM::Torus(torus, reflSolid));
279  }
280 
281  G4Trap* trap = dynamic_cast<G4Trap*>(consSolid);
282  if (trap) {
283  return Register(new Geant4GM::Trap(trap, reflSolid));
284  }
285 
286  G4Trd* trd = dynamic_cast<G4Trd*>(consSolid);
287  if (trd) {
288  return Register(new Geant4GM::Trd(trd, reflSolid));
289  }
290 
291  G4Tubs* tubs = dynamic_cast<G4Tubs*>(consSolid);
292  if (tubs) {
293  return Register(new Geant4GM::Tubs(tubs, reflSolid));
294  }
295 
296  G4DisplacedSolid* displaced = dynamic_cast<G4DisplacedSolid*>(consSolid);
297  if (displaced) {
298  ImportSolid(displaced->GetConstituentMovedSolid());
299  return Register(new Geant4GM::DisplacedSolid(displaced, reflSolid));
300  }
301 
302  G4ScaledSolid* scaled = dynamic_cast<G4ScaledSolid*>(consSolid);
303  if (scaled) {
304  ImportSolid(scaled->GetUnscaledSolid());
305  return Register(new Geant4GM::ScaledSolid(scaled, reflSolid));
306  }
307 
308  G4BooleanSolid* boolean = dynamic_cast<G4BooleanSolid*>(consSolid);
309  if (boolean) {
310  ImportConstituentSolid(0, boolean);
311  ImportConstituentSolid(1, boolean);
312  VGM::IBooleanSolid* vgmBoolean =
313  new Geant4GM::BooleanSolid(boolean, reflSolid);
314  Register(vgmBoolean);
315 
316  if (Debug() > 0) {
317  BaseVGM::DebugInfo();
318  std::cout << "Imported Boolean solid: ";
319  if (Debug() > 1) std::cout << vgmBoolean;
320  std::cout << std::endl;
321  BaseVGM::DebugInfo();
322  std::cout << *vgmBoolean << std::endl;
323  }
324 
325  return vgmBoolean;
326  }
327 
328  G4MultiUnion* multiUnion = dynamic_cast<G4MultiUnion*>(consSolid);
329  if (multiUnion) {
330  for (G4int i = 0; i < multiUnion->GetNumberOfSolids(); ++i) {
331  ImportConstituentSolid(i, multiUnion);
332  }
333  VGM::IMultiUnion* vgmMultiUnion =
334  new Geant4GM::MultiUnion(multiUnion, reflSolid);
335  Register(vgmMultiUnion);
336 
337  if (Debug() > 0) {
338  BaseVGM::DebugInfo();
339  std::cout << "Imported MultiUnion solid: ";
340  if (Debug() > 1) std::cout << vgmMultiUnion;
341  std::cout << std::endl;
342  BaseVGM::DebugInfo();
343  std::cout << *vgmMultiUnion << std::endl;
344  }
345 
346  return vgmMultiUnion;
347  }
348 
349  std::cerr << "Geant4GM::Factory::ImportSolid: " << std::endl;
350  std::cerr << "Unsupported solid type (solid \"" << solid->GetName() << "\""
351  << " type \"" << solid->GetEntityType() << "\")" << std::endl;
352 
353  if (Ignore()) {
354  std::cerr << "*** Warning: Using a box instead ***" << std::endl;
355  return Register(new Geant4GM::Box(solid->GetName(),
356  DummyBoxDimensions() / ClhepVGM::Units::Length(),

359  }
360  else {

362  exit(1);
363  }

554 VGM::IPlacement* Geant4GM::Factory::ImportPVPair(VGM::IVolume* volume,
555  VGM::IVolume* motherVolume, G4PhysicalVolumesPair pvPair)

562  VGM::IVolume* iv1 = Geant4GM::VolumeMap::Instance()->GetVolume(lv1);
563  if (!iv1) {
564  // a new logical volume has been created by G4 reflection factory
565  // has to be imported into VGM
566  iv1 = ImportLV(lv1, volume->MediumName());
567  ImportDaughters(lv1);
568  ImportPositions(lv1);
569  }
570  VGM::IPlacement* placement1 = new Geant4GM::Placement(iv1, motherVolume, pv1);
571  // Register physical volume in the map
572  Geant4GM::PlacementMap::Instance()->AddPlacement(placement1, pv1);
573 
574  G4VPhysicalVolume* pv2 = pvPair.second;
575  if (pv2) {
576  G4LogicalVolume* g4MotherLV =
577  Geant4GM::VolumeMap::Instance()->GetVolume(motherVolume);
578

582  if (g4ReflectionFactory->IsReflected(g4MotherLV))
583  mlv2 = g4ReflectionFactory->GetConstituentLV(g4MotherLV);
584  if (g4ReflectionFactory->IsConstituent(g4MotherLV))
585  mlv2 = g4ReflectionFactory->GetReflectedLV(g4MotherLV);

597  // should not happen

599  std::cerr << " Missing mapping of existing LV to VGM" << std::endl;
600  std::cerr << "*** Error: Aborting execution ***" << std::endl;
601  exit(1);
602  }
603 
604  G4LogicalVolume* lv2 = pv2->GetLogicalVolume();
605  VGM::IVolume* iv2 = Geant4GM::VolumeMap::Instance()->GetVolume(lv2);
606  if (!iv2) {
607  // a new logical volume has been created by G4 reflection factory

610  ImportDaughters(lv2);
