44 double BaseVGM::VPolyhedra::ConvertRadiusFactor() const
45 {
46  //
47  double phiTotal = DeltaPhi() / 180. * M_PI;
48  if ((phiTotal <= 0) || (phiTotal >= 2 * M_PI * (1 - 1e-16)))
49  phiTotal = 2 * M_PI;
50 
51  return cos(0.5 * phiTotal / NofSides());
52 }
