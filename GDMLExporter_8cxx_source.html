106 void XmlVGM::GDMLExporter::GenerateSection(VGM::IVolume* volume)
107 {
111 
112  // Create section

115 
116  // Process basic elements needed by geometry tree

134 void XmlVGM::GDMLExporter::ProcessVolume(VGM::IVolume* volume)
135 {
137 
138  int nofDaughters = volume->NofDaughters();
139 
140  if (nofDaughters == 0) {
141  // Open composition
142  fWriter->OpenComposition(volume->Name(), volume->MaterialName());
143  }
144  else {
145 
146  // Recursively process daughters
147  //
148  for (int i = 0; i < nofDaughters; i++) {
149 
150  VGM::IVolume* dVolume = volume->Daughter(i)->Volume();
151 
152  if (fVolumeNames.find(dVolume->Name()) == fVolumeNames.end())
153  ProcessVolume(dVolume);
154  }
155 
156  // Write the volume with its childs now
157  //
158  // Open composition
159  fWriter->OpenComposition(volume->Name(), volume->MaterialName());
160 
161  // Write positions
162  for (int j = 0; j < nofDaughters; j++) {
163 
164  if (fDebug > 1) {
165  std::cout << "processing " << j << "th daughter of " << volume->Name()
166  << std::endl;
167  }
168 
169  fWriter->WritePlacement(*volume->Daughter(j));
170  }
171  }
172 
173  // Close composition
174  fWriter->CloseComposition();
175  fWriter->WriteEmptyLine();
176 
177  // store the name of logical volume in the set
178  fVolumeNames.insert(fVolumeNames.begin(), volume->Name());
179 }
