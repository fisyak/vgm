47

52  switch (boolType) {

54  fBooleanSolid = new G4IntersectionSolid(

58  case VGM::kSubtraction:

66  break;
67 
68  case VGM::kUnknownBoolean:
69  default:
70  std::cerr << " Geant4GM::BooleanSolid::BooleanSolid:" << std::endl;
71  std::cerr << " Unknown Boolean solid type (solid \"" << name << "\")"

74  exit(1);
