151 VGM::ISolid* BaseVGM::VFactory::ExportBooleanSolid(
152  VGM::IBooleanSolid* solid, VGM::IFactory* factory) const
153 
154 {
155  // Exports specified Boolean solid to given factory
156  // ---
157 
158  // Export constituent solids first
159  VGM::ISolid* solidA = ExportSolid(solid->ConstituentSolidA(), factory);
160  VGM::ISolid* solidB = ExportSolid(solid->ConstituentSolidB(), factory);
161  // Can lead to a duplication of solids in case
162  // the solid has been already exported
163  // Should not harm, but will be better to be avoided
164 
165  VGM::BooleanType boolType = solid->BoolType();
166  VGM::Transform transform = solid->Displacement();
167 
168  VGM::ISolid* newSolid = 0;
169  if (boolType == VGM::kIntersection) {
170  newSolid = factory->CreateIntersectionSolid(
171  solid->Name(), solidA, solidB, transform);
172  }
173  else if (boolType == VGM::kSubtraction) {
174  newSolid =
175  factory->CreateSubtractionSolid(solid->Name(), solidA, solidB, transform);
176  }
177  else if (boolType == VGM::kUnion) {
178  newSolid =
179  factory->CreateUnionSolid(solid->Name(), solidA, solidB, transform);
180  }
181 
182  if (!newSolid) {
183  std::cerr << " BaseVGM::VFactory::ExportBooleanSolid:" << std::endl;
184  std::cerr << " Unknown Boolean type (solid \"" << solid->Name() << "\")"
185  << std::endl;
186  std::cerr << "*** Error: Aborting execution ***" << std::endl;
187  exit(1);
188  }
189 
190  return newSolid;
191 }

214 VGM::ISolid* BaseVGM::VFactory::ExportSolid(
215  VGM::ISolid* solid, VGM::IFactory* factory) const
216 {
217  // Exports specified solid to given factory
218  // ---
219 
220  if (Debug() > 0) {
221  BaseVGM::DebugInfo();
222  std::cout << " Exporting solid: ";
223  if (Debug() > 1) std::cout << solid;
224  std::cout << std::endl;
225  BaseVGM::DebugInfo();
226  std::cout << " " << *solid << std::endl;
227  }
228 
229  VGM::SolidType solidType = solid->Type();
230  if (solidType == VGM::kArb8) {
231  VGM::IArb8* arb8 = dynamic_cast<VGM::IArb8*>(solid);
232  std::vector<VGM::TwoVector> vertices;
233  for (int i = 0; i < arb8->NofVertices(); ++i)
234  vertices.push_back(arb8->Vertex(i));
235  return factory->CreateArb8(arb8->Name(), arb8->ZHalfLength(), vertices);
236  }
237  else if (solidType == VGM::kBox) {
238  VGM::IBox* box = dynamic_cast<VGM::IBox*>(solid);
239  return factory->CreateBox(
240  box->Name(), box->XHalfLength(), box->YHalfLength(), box->ZHalfLength());
241  }
242  else if (solidType == VGM::kCons) {
243  VGM::ICons* cons = dynamic_cast<VGM::ICons*>(solid);
244  return factory->CreateCons(cons->Name(), cons->InnerRadiusMinusZ(),
245  cons->OuterRadiusMinusZ(), cons->InnerRadiusPlusZ(),
246  cons->OuterRadiusPlusZ(), cons->ZHalfLength(), cons->StartPhi(),
247  cons->DeltaPhi());
248  }
249  else if (solidType == VGM::kCtubs) {
250  VGM::ICtubs* ctubs = dynamic_cast<VGM::ICtubs*>(solid);
251  return factory->CreateCtubs(ctubs->Name(), ctubs->InnerRadius(),
252  ctubs->OuterRadius(), ctubs->ZHalfLength(), ctubs->StartPhi(),
253  ctubs->DeltaPhi(), ctubs->NxLow(), ctubs->NyLow(), ctubs->NzLow(),
254  ctubs->NxHigh(), ctubs->NyHigh(), ctubs->NzHigh());
255  }
256  else if (solidType == VGM::kEllipsoid) {
257  VGM::IEllipsoid* ellipsoid = dynamic_cast<VGM::IEllipsoid*>(solid);
258  return factory->CreateEllipsoid(ellipsoid->Name(), ellipsoid->XSemiAxis(),
259  ellipsoid->YSemiAxis(), ellipsoid->ZSemiAxis(), ellipsoid->ZBottomCut(),
260  ellipsoid->ZTopCut());
261  }
262  else if (solidType == VGM::kEltu) {
263  VGM::IEllipticalTube* eltu = dynamic_cast<VGM::IEllipticalTube*>(solid);
264  return factory->CreateEllipticalTube(
265  eltu->Name(), eltu->Dx(), eltu->Dy(), eltu->ZHalfLength());
266  }
267  else if (solidType == VGM::kExtruded) {
268  VGM::IExtrudedSolid* xtru = dynamic_cast<VGM::IExtrudedSolid*>(solid);
269  std::vector<VGM::TwoVector> polygon;
270  for (int i = 0; i < xtru->NofVertices(); ++i)
271  polygon.push_back(xtru->Vertex(i));
272  std::vector<std::vector<double> > zsections;
273  for (int i = 0; i < xtru->NofZSections(); ++i) {
274  std::vector<double> zsection;
275  zsection.push_back(xtru->ZPosition(i));
276  zsection.push_back(xtru->Offset(i).first);
277  zsection.push_back(xtru->Offset(i).second);
278  zsection.push_back(xtru->Scale(i));
279  zsections.push_back(zsection);
280  }
281  return factory->CreateExtrudedSolid(xtru->Name(), polygon, zsections);
282  }
283  else if (solidType == VGM::kHype) {
284  VGM::IHype* hype = dynamic_cast<VGM::IHype*>(solid);
285  return factory->CreateHype(hype->Name(), hype->InnerRadius(),
286  hype->OuterRadius(), hype->InnerStereoAngle(), hype->OuterStereoAngle(),
287  hype->ZHalfLength());
288  }
289  else if (solidType == VGM::kPara) {
290  VGM::IPara* para = dynamic_cast<VGM::IPara*>(solid);
291  return factory->CreatePara(para->Name(), para->XHalfLength(),
292  para->YHalfLength(), para->ZHalfLength(), para->Alpha(), para->Theta(),
293  para->Phi());
294  }
295  else if (solidType == VGM::kParaboloid) {
296  VGM::IParaboloid* paraboloid = dynamic_cast<VGM::IParaboloid*>(solid);
297  return factory->CreateParaboloid(paraboloid->Name(),
298  paraboloid->RadiusMinusZ(), paraboloid->RadiusPlusZ(),
299  paraboloid->ZHalfLength());
300  }
301  else if (solidType == VGM::kPolycone) {
302  VGM::IPolycone* polycone = dynamic_cast<VGM::IPolycone*>(solid);
303  return factory->CreatePolycone(polycone->Name(), polycone->StartPhi(),
304  polycone->DeltaPhi(), polycone->NofZPlanes(), polycone->ZValues(),
305  polycone->InnerRadiusValues(), polycone->OuterRadiusValues());
306  }
307  else if (solidType == VGM::kPolyhedra) {
308  VGM::IPolyhedra* polyhedra = dynamic_cast<VGM::IPolyhedra*>(solid);
309  return factory->CreatePolyhedra(polyhedra->Name(), polyhedra->StartPhi(),
310  polyhedra->DeltaPhi(), polyhedra->NofSides(), polyhedra->NofZPlanes(),
311  polyhedra->ZValues(), polyhedra->InnerRadiusValues(),
312  polyhedra->OuterRadiusValues());
313  }
314  else if (solidType == VGM::kSphere) {
315  VGM::ISphere* sphere = dynamic_cast<VGM::ISphere*>(solid);
316  return factory->CreateSphere(sphere->Name(), sphere->InnerRadius(),
317  sphere->OuterRadius(), sphere->StartPhi(), sphere->DeltaPhi(),
318  sphere->StartTheta(), sphere->DeltaTheta());
319  }
320  else if (solidType == VGM::kTessellated) {
321  VGM::ITessellatedSolid* tessellated =
322  dynamic_cast<VGM::ITessellatedSolid*>(solid);
323 
324  std::vector<std::vector<VGM::ThreeVector> > facets;
325  for (int i = 0; i < tessellated->NofFacets(); ++i) {
326  std::vector<VGM::ThreeVector> facet;
327  for (int j = 0; j < tessellated->NofVertices(i); ++j) {
328  facet.push_back(tessellated->Vertex(i, j));
329  }
330  facets.push_back(facet);
331  }
332  return factory->CreateTessellatedSolid(tessellated->Name(), facets);
333  }
334  else if (solidType == VGM::kTorus) {
335  VGM::ITorus* torus = dynamic_cast<VGM::ITorus*>(solid);
336  return factory->CreateTorus(torus->Name(), torus->InnerRadius(),
337  torus->OuterRadius(), torus->AxialRadius(), torus->StartPhi(),
338  torus->DeltaPhi());
339  }
340  else if (solidType == VGM::kTrap) {
341  VGM::ITrap* trap = dynamic_cast<VGM::ITrap*>(solid);
342  return factory->CreateTrap(trap->Name(), trap->ZHalfLength(), trap->Theta(),
343  trap->Phi(), trap->YHalfLengthMinusZ(), trap->XHalfLengthMinusZMinusY(),
344  trap->XHalfLengthMinusZPlusY(), trap->AlphaMinusZ(),
345  trap->YHalfLengthPlusZ(), trap->XHalfLengthPlusZMinusY(),
346  trap->XHalfLengthPlusZPlusY(), trap->AlphaPlusZ());
347  }
348  else if (solidType == VGM::kTrd) {
349  VGM::ITrd* trd = dynamic_cast<VGM::ITrd*>(solid);
350  return factory->CreateTrd(trd->Name(), trd->XHalfLengthMinusZ(),
351  trd->XHalfLengthPlusZ(), trd->YHalfLengthMinusZ(),
352  trd->YHalfLengthPlusZ(), trd->ZHalfLength());
353  }
354  else if (solidType == VGM::kTubs) {
355  VGM::ITubs* tubs = dynamic_cast<VGM::ITubs*>(solid);
356  return factory->CreateTubs(tubs->Name(), tubs->InnerRadius(),
357  tubs->OuterRadius(), tubs->ZHalfLength(), tubs->StartPhi(),
358  tubs->DeltaPhi());
359  }
360  else if (solidType == VGM::kDisplaced) {
361  VGM::IDisplacedSolid* displaced =
362  dynamic_cast<VGM::IDisplacedSolid*>(solid);
363  return ExportDisplacedSolid(displaced, factory);
364  }
365  else if (solidType == VGM::kScaled) {
366  VGM::IScaledSolid* scaled = dynamic_cast<VGM::IScaledSolid*>(solid);
367  return ExportScaledSolid(scaled, factory);
368  }
369  else if (solidType == VGM::kBoolean) {
370  VGM::IBooleanSolid* boolean = dynamic_cast<VGM::IBooleanSolid*>(solid);
371  return ExportBooleanSolid(boolean, factory);
372  }
373  else if (solidType == VGM::kMultiUnion) {
374  VGM::IMultiUnion* multiUnion = dynamic_cast<VGM::IMultiUnion*>(solid);
375  return ExportMultiUnion(multiUnion, factory);
376  }
377 
378  std::cerr << " BaseVGM::VFactory::ExportSolid:" << std::endl;
379  std::cerr << " Unknown solid type (solid \"" << solid->Name() << "\")"
380  << std::endl;
381  std::cerr << "*** Error: Aborting execution ***" << std::endl;
382  exit(1);
383  return 0;
384 }

389 {
390  // Exports all volumes.

397

399 
400  for (unsigned int i = 0; i < Volumes().size(); i++) {
401 
402  VGM::IVolume* volume = Volumes()[i];
403 
404  // Export solid

413 
414  // Debug info

421  if (Debug() > 1)
422  std::cout << " "
423  << " oldVGM=" << volume << " newVGM=" << newVolume;
424  std::cout << std::endl;
425  }
426  }
427 
428  return volumeMap;
429 }

442  }
443 #endif
444 
445  VGM::IVolume* newVolume = (*volumeMap)[placement->Volume()];
446  VGM::IVolume* newMother = (*volumeMap)[placement->Mother()];
447 
448  // If boolean or scaled solid that have to be reflected
450  VGM::Transform transform = placement->Transformation();
451  VGM::IBooleanSolid* booleanSolid =
452  dynamic_cast<VGM::IBooleanSolid*>(placement->Volume()->Solid());
453  VGM::IScaledSolid* scaledSolid =
454  dynamic_cast<VGM::IScaledSolid*>(placement->Volume()->Solid());
455  if ((booleanSolid && booleanSolid->ToBeReflected()) ||
456  (scaledSolid && scaledSolid->ToBeReflected())) {
457  transform[VGM::kReflZ] = 1;
458  }
459 
460  VGM::IPlacement* newPlacement = factory->CreatePlacement(
461  placement->Name(), placement->CopyNo(), newVolume, newMother, transform);
462 
463  return newPlacement;
464 }

500 void BaseVGM::VFactory::ExportPlacements(

502 {
503  // Exports all placements.
504  // ---
505 
506 #ifdef NEW_DEBUG
507  if (Debug() > 0) {
508  BaseVGM::DebugInfo();
509  std::cout << "Exporting placements:" << std::endl;
510  }
511 #endif
512 
513  for (unsigned int i = 0; i < Volumes().size(); i++) {
514 
515  VGM::IVolume* volume = Volumes()[i];
516 
517 #ifndef NEW_DEBUG

519  BaseVGM::DebugInfo();
520  std::cout << "ExportPlacements for " << i << "th volume ";
521  if (Debug() > 1) std::cout << volume << " ";
522  std::cout << volume->Name() << std::endl;
523  }
524 #endif
525 
526  for (int id = 0; id < volume->NofDaughters(); id++) {
527 
528  VGM::IPlacement* daughter = volume->Daughter(id);
529 
530 #ifndef NEW_DEBUG
531  if (Debug() > 0) {
532  BaseVGM::DebugInfo();
533  std::cout << " " << id << "th daughter vol = ";
534  if (Debug() > 1) std::cout << daughter->Volume() << " ";
535  std::cout << daughter->Volume()->Name();
536  }
537 #endif
538 
539 #ifdef NEW_DEBUG
540  if (Debug() > 0) {
541  BaseVGM::DebugInfo();
542  std::cout << "Exporting placement: ";
543  if (Debug() > 1) std::cout << daughter;
544  std::cout << std::endl;
545  BaseVGM::DebugInfo();
546  std::cout << " " << *daughter << std::endl;
547  }
548 #endif
549 
550  if (daughter->Type() == VGM::kSimplePlacement) {
551  ExportSimplePlacement(daughter, factory, volumeMap);
552  }
553  else if (daughter->Type() == VGM::kMultiplePlacement) {
554  ExportMultiplePlacement(daughter, factory, volumeMap);
555  }
556  else {
557  std::cout << std::endl;
558  std::cerr << " BaseVGM::VFactory::ExportPlacements:" << std::endl;
559  std::cerr << " Unknown placement type (placement \""
560  << daughter->Name() << "\")" << std::endl;
561  std::cerr << "*** Error: Aborting execution ***" << std::endl;
562  exit(1);
563  }
564  }
565  }
566 
567  // Position the top volume
568  // (top volume has no mother volume that's why it must be placed
569  // explicitely)
570 
571  VGM::IVolume* topVolume = (*volumeMap)[Top()->Volume()];
572 
573  factory->CreatePlacement(
574  Top()->Name(), Top()->CopyNo(), topVolume, 0, Identity());
575 
576  delete volumeMap;
577 }

593 //_____________________________________________________________________________
594 bool BaseVGM::VFactory::Export(VGM::IFactory* factory) const
595 {
597 
598  // set the mode to destination factory
599  factory->SetSingleMode(SingleMode());
600 
601  if (!SingleMode()) {
602  // Export materials
603  //
604  fMaterialFactory->Export(factory->MaterialFactory());
605 
606  // Export volumes
607  //
608  VolumeMap* volumeMap = ExportVolumeStore(factory);
609  ExportPlacements(factory, volumeMap);
610 
611  return true;
612  }
613  else {
614  // one solid mode
615 
616  // Check if a solid was created/imported
617  if (!SingleSolid()) {
618  std::cerr << "++ Warning: ++ " << std::endl;
619  std::cerr << " BaseVGM::Export:" << std::endl;
620  std::cerr << " A solid must be created/imported first." << std::endl;
621 
622  return false;
623  }
624 
625  // Export solid
626  VGM::ISolid* solid = ExportSolid(SingleSolid(), factory);
627  factory->SetSolid(solid);
628 
629  return (solid != 0);
630  }
631 }
