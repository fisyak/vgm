179 bool Geant4GM::Placement::MultiplePlacementData(VGM::Axis& axis, int& nofItems,
180  double& width, double& offset, double& halfGap) const
181 {
182  // Fill data only if multiple placement
183  if (Type() != VGM::kMultiplePlacement) return false;
184 
185  // Different get functions for PVReplica and PVDivisions
186  //
187  EAxis g4Axis = kUndefined;
188  if (dynamic_cast<G4PVReplica*>(fPhysicalVolume)) {
189  bool consuming;
190 
191  G4double offset0;
192  fPhysicalVolume->GetReplicationData(
193  g4Axis, nofItems, width, offset0, consuming);
194 
195  // Different meaning of offset in R/Phi division
196  // for replica than for division:
197  // the offset includes start value (rmin, sphi)
198 
199  G4double xlo = 0.;
200  VGM::ISolid* solid = Volume()->Solid();
201 
202  if (g4Axis == kRho) {
203  if (solid->Type() == VGM::kTubs)
204  xlo = dynamic_cast<VGM::ITubs*>(solid)->InnerRadius();
205  }
206 
207  if (g4Axis == kPhi) {
208  // Different meaning of offset in R/Phi division
209  // for replica than for division:
210  // the offset includes start value (rmin, sphi)
211 
212  if (solid->Type() == VGM::kCons)
213  xlo = dynamic_cast<VGM::ICons*>(solid)->StartPhi();
214 
215  if (solid->Type() == VGM::kTubs)
216  xlo = dynamic_cast<VGM::ITubs*>(solid)->StartPhi();
217 
218  if (solid->Type() == VGM::kPolycone)
219  xlo = dynamic_cast<VGM::IPolycone*>(solid)->StartPhi();
220 
221  if (solid->Type() == VGM::kPolyhedra)
222  xlo = dynamic_cast<VGM::IPolyhedra*>(solid)->StartPhi();
223  }
224 
225  offset = offset0 - xlo;
226  }
227  else if (dynamic_cast<G4PVDivision*>(fPhysicalVolume) ||
228  dynamic_cast<G4ReplicatedSlice*>(fPhysicalVolume)) {
229  G4VDivisionParameterisation* param =
230  dynamic_cast<G4VDivisionParameterisation*>(
231  fPhysicalVolume->GetParameterisation());
232 
233  if (!param) {
234  std::cerr << " Geant4GM::Placement::MultiplePlacementData: "
235  << std::endl;
236  std::cerr << " Incorrect parameterisation type for "
237  "G4PVDivision/G4ReplicatedSlice"
238  << std::endl;
239  std::cerr << " (G4VDivisionParameterisation type was expected.)"
240  << std::endl;
241  std::cerr << "*** Error: Aborting execution ***" << std::endl;
242  exit(1);
243  }
244 
245  g4Axis = param->GetAxis();
246  nofItems = param->GetNoDiv();
247  width = param->GetWidth();
248  offset = param->GetOffset();
249  halfGap = param->GetHalfGap();
250  }
251 
252  axis = GetAxis(g4Axis);
253 
254  // Convert units
255  offset *= ClhepVGM::Units::AxisUnit(axis);
256  width *= ClhepVGM::Units::AxisUnit(axis);
257  halfGap *= ClhepVGM::Units::AxisUnit(axis);
258 
259  return true;
260 }
