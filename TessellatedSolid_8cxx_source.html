36 Geant4GM::TessellatedSolid::TessellatedSolid(
37  const std::string& name, std::vector<std::vector<VGM::ThreeVector> > facets)
38  : VGM::ISolid(),
39  VGM::ITessellatedSolid(),
40  BaseVGM::VTessellatedSolid(),
41  fName(name),
42  fTessellatedSolid(0)
43 {
47 
48  fTessellatedSolid = new G4TessellatedSolid();
49 
50  // Add triangular facets
51  //
52  for (G4int i = 0; i < G4int(facets.size()); i++) {
53 
54  std::vector<VGM::ThreeVector> facet = facets[i];
55 
56  // check number of vertices
57  if (facet.size() != 3 && facet.size() != 4) {
58  std::cerr << "+++ Error +++" << std::endl;
59  std::cerr << " Number of vertices in a facet = " << facet.size()
60  << " has to be == 3 or 4" << std::endl;
61  exit(1);
62  }
63 
64  if (facet.size() == 3) {
65  VGM::ThreeVector vertex0 = facet[0];
66  VGM::ThreeVector vertex1 = facet[1];
67  VGM::ThreeVector vertex2 = facet[2];
68 
69  fTessellatedSolid->AddFacet(new G4TriangularFacet(
70  G4ThreeVector(vertex0[VGM::kDx] / ClhepVGM::Units::Length(),
71  vertex0[VGM::kDy] / ClhepVGM::Units::Length(),
72  vertex0[VGM::kDz] / ClhepVGM::Units::Length()),
73  G4ThreeVector(vertex1[VGM::kDx] / ClhepVGM::Units::Length(),
74  vertex1[VGM::kDy] / ClhepVGM::Units::Length(),
75  vertex1[VGM::kDz] / ClhepVGM::Units::Length()),
76  G4ThreeVector(vertex2[VGM::kDx] / ClhepVGM::Units::Length(),
77  vertex2[VGM::kDy] / ClhepVGM::Units::Length(),
78  vertex2[VGM::kDz] / ClhepVGM::Units::Length()),
79  ABSOLUTE));
80  }
81  }
82 
83  // Add quadrangular facets
84  //
85  for (G4int i = 0; i < G4int(facets.size()); i++) {
86 
87  std::vector<VGM::ThreeVector> facet = facets[i];
88 
89  if (facet.size() == 4) {
90  VGM::ThreeVector vertex0 = facet[0];
91  VGM::ThreeVector vertex1 = facet[1];
92  VGM::ThreeVector vertex2 = facet[2];
93  VGM::ThreeVector vertex3 = facet[3];
94 
95  fTessellatedSolid->AddFacet(new G4QuadrangularFacet(
96  G4ThreeVector(vertex0[VGM::kDx] / ClhepVGM::Units::Length(),
97  vertex0[VGM::kDy] / ClhepVGM::Units::Length(),
98  vertex0[VGM::kDz] / ClhepVGM::Units::Length()),
99  G4ThreeVector(vertex1[VGM::kDx] / ClhepVGM::Units::Length(),
100  vertex1[VGM::kDy] / ClhepVGM::Units::Length(),
101  vertex1[VGM::kDz] / ClhepVGM::Units::Length()),
102  G4ThreeVector(vertex2[VGM::kDx] / ClhepVGM::Units::Length(),
103  vertex2[VGM::kDy] / ClhepVGM::Units::Length(),
104  vertex2[VGM::kDz] / ClhepVGM::Units::Length()),
105  G4ThreeVector(vertex3[VGM::kDx] / ClhepVGM::Units::Length(),
106  vertex3[VGM::kDy] / ClhepVGM::Units::Length(),
107  vertex3[VGM::kDz] / ClhepVGM::Units::Length()),
108  ABSOLUTE));
109  }
110  }
111 
112  fTessellatedSolid->SetSolidClosed(true);
113 
114  // G4cout << *fTessellatedSolid << G4endl;
115 
116  Geant4GM::SolidMap::Instance()->AddSolid(this, fTessellatedSolid);
117 }

171 void Geant4GM::TessellatedSolid::CheckFacetIndex(int ifacet) const
172 {
173  if (ifacet < 0 || ifacet > NofFacets()) {
174  std::cerr << "+++ Error +++" << std::endl;
175  std::cerr << " Wrong facet index: " << ifacet << std::endl;
176  exit(1);
177  }
178 
179  G4VFacet* facet = fTessellatedSolid->GetFacet(ifacet);

182  std::cerr << " Facet with index: " << ifacet << " not found."

225 {
226  CheckVertexIndex(ifacet, index);

228  G4VFacet* facet = fTessellatedSolid->GetFacet(ifacet);

232  vertex.push_back(facet->GetVertex(index).y() * ClhepVGM::Units::Length());
233  vertex.push_back(facet->GetVertex(index).z() * ClhepVGM::Units::Length());
